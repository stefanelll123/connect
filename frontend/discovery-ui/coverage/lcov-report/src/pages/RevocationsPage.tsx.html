function formatDate(iso: string) {
  return new Date(iso).toLocaleString(undefined, {
    year: "numeric",
    month: "short",
    day: "numeric",
    hour: "2-digit",
    minute: "2-digit",
  });
}

const RevokedCredentialsTab: React.FC = () => {
  const { data, isLoading: loading } = useQuery(
    ["revoked-credentials"],
    () => listCredentials({ status: "revoked", limit: 100 }),
  );
 
  const credentials: Credential[] = data ?? [];
 
  return (
    <TableContainer component={Paper}>
      <Table size="small">
        <TableHead>
          <TableRow>
            <TableCell>Credential ID</TableCell>
            <TableCell>Type</TableCell>
            <TableCell>Subject</TableCell>
            <TableCell>Env</TableCell>
            <TableCell>Revoked At</TableCell>
            <TableCell>Status</TableCell>
          </TableRow>
        </TableHead>
        <TableBody>
          {loading && (
            <TableRow><TableCell colSpan={6} align="center">Loading…</TableCell></TableRow>
          )}
          {!loading && credentials.length === 0 && (
            <TableRow><TableCell colSpan={6} align="center">No revoked credentials.</TableCell></TableRow>
          )}
          {credentials.map((cred) => (
            <TableRow key={cred.id} hover>
              <TableCell>
                <Tooltip title={cred.id}>
                  <code>{cred.id.slice(0, 12)}</code>
                </Tooltip>
              </TableCell>
              <TableCell><Chip label={cred.credential_type} size="small" /></TableCell>
              <TableCell>
                <Tooltip title={cred.subject_did}>
                  <code>{cred.subject_did.slice(-8)}</code>
                </Tooltip>
              </TableCell>
              <TableCell>{cred.env}</TableCell>
              <TableCell>{cred.revoked_at ? formatDate(cred.revoked_at) : "—"}</TableCell>
              <TableCell><StatusChip status={cred.status} /></TableCell>
            </TableRow>
          ))}
        </TableBody>
      </Table>
    </TableContainer>
  );
};
 
// ── Status Lists Tab ─────────────────────────────────────────────────────────
interface StatusListDetailProps {
  statusList: StatusList;
  onClose: () => void;
}
 
const StatusListDetail: React.FC<StatusListDetailProps> = ({ statusList, onClose }) => {
  const revokedPct =
    statusList.total_entries > 0
      ? Math.round((statusList.revoked_count / statusList.total_entries) * 100)
      : 0;
 
  return (
    <Drawer anchor="right" open onClose={onClose} PaperProps={{ sx: { width: 420, p: 3 } }}>
      <Typography variant="h6" gutterBottom>Status List Detail</Typography>
      <Divider sx={{ mb: 2 }} />
      <Typography variant="body2" sx={{ mb: 1 }}>
        <strong>ID:</strong> <code>{statusList.id}</code>
      </Typography>
      <Typography variant="body2" sx={{ mb: 1 }}>
        <strong>Environment:</strong> {statusList.issued_for_env}
      </Typography>
      <Typography variant="body2" sx={{ mb: 1 }}>
        <strong>Last Published:</strong>{" "}
        {statusList.last_published_at ? formatDate(statusList.last_published_at) : "—"}
      </Typography>
      <Typography variant="body2" sx={{ mb: 1 }}>
        <strong>Anchor Hash:</strong>{" "}
        <Tooltip title={statusList.anchor_hash ?? ""}>
          <code>{(statusList.anchor_hash ?? "").slice(0, 12)}</code>
        </Tooltip>
      </Typography>
      <Divider sx={{ my: 2 }} />
      <Typography variant="subtitle2" gutterBottom>Allocation</Typography>
      <Box sx={{ mb: 1 }}>
        <Box sx={{ display: "flex", justifyContent: "space-between", mb: 0.5 }}>
          <Typography variant="body2">Revoked</Typography>
          <Typography variant="body2">{statusList.revoked_count} / {statusList.total_entries} ({revokedPct}%)</Typography>
        </Box>
        <LinearProgress variant="determinate" value={revokedPct} color="error" />
      </Box>
      <Typography variant="body2">
        Active: {statusList.active_count}
      </Typography>
    </Drawer>
  );
};
 
const StatusListsTab: React.FC = () => {
  const { data, isLoading: loading } = useQuery(["status-lists"], listStatusLists);
  const [selected, setSelected] = useState<StatusList | null>(null);
 
  const lists: StatusList[] = data ?? [];
 
  return (
    <>
      <TableContainer component={Paper}>
        <Table size="small">
          <TableHead>
            <TableRow>
              <TableCell>Status List ID</TableCell>
              <TableCell>Env</TableCell>
              <TableCell>Total</TableCell>
              <TableCell>Revoked</TableCell>
              <TableCell>Active</TableCell>
              <TableCell>Last Published</TableCell>
              <TableCell>Anchor Hash</TableCell>
            </TableRow>
          </TableHead>
          <TableBody>
            {loading && (
              <TableRow><TableCell colSpan={7} align="center">Loading…</TableCell></TableRow>
            )}
            {!loading && lists.length === 0 && (
              <TableRow><TableCell colSpan={7} align="center">No status lists found.</TableCell></TableRow>
            )}
            {lists.map((sl) => (
              <TableRow
                key={sl.id}
                hover
                sx={{ cursor: "pointer" }}
                onClick={() => setSelected(sl)}
              >
                <TableCell><code>{sl.id}</code></TableCell>
                <TableCell>{sl.issued_for_env}</TableCell>
                <TableCell>{sl.total_entries}</TableCell>
                <TableCell>{sl.revoked_count}</TableCell>
                <TableCell>{sl.active_count}</TableCell>
                <TableCell>{sl.last_published_at ? formatDate(sl.last_published_at) : "—"}</TableCell>
                <TableCell>
                  <Tooltip title={sl.anchor_hash ?? ""}>
                    <code>{(sl.anchor_hash ?? "").slice(0, 12)}</code>
                  </Tooltip>
                </TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>
      </TableContainer>
      {selected && (
        <StatusListDetail statusList={selected} onClose={() => setSelected(null)} />
      )}
    </>
  );
};
